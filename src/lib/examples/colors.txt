<script>
    let selected = $state('')
    let selectedHex = $state('')

    const groups = {
        reds: ['#ff0000', '#dc143c', '#b22222', '#8b0000', '#cd5c5c', '#f08080'],
        oranges: ['#ff4500', '#ff6347', '#ff7f50', '#ff8c00', '#ffa500'],
        yellows: ['#ffd700', '#ffff00', '#f0e68c', '#bdb76b'],
        greens: ['#00ff00', '#32cd32', '#228b22', '#006400', '#2e8b57', '#3cb371'],
        cyans: ['#00ffff', '#40e0d0', '#20b2aa', '#008b8b', '#008080'],
        blues: ['#0000ff', '#1e90ff', '#4169e1', '#4682b4', '#000080', '#191970'],
        purples: ['#8a2be2', '#9400d3', '#9370db', '#663399', '#4b0082', '#800080'],
        pinks: ['#ff1493', '#ff69b4', '#ffc0cb', '#db7093', '#c71585'],
    }
</script>

<div class="picker">
    <div class="preview" style="background-color: {selectedHex || '#333'};">
        <span class="preview-text">{selectedHex || 'pick a colour'}</span>
    </div>

    {#each Object.entries(groups) as [name, colors]}
        <div class="group">
            <span class="label">{name}</span>
            <div class="swatches">
                {#each colors as hex}
                    <button
                        class="swatch"
                        class:selected={selectedHex === hex}
                        style="background-color: {hex};"
                        onclick={() => { selectedHex = hex; selected = name }}
                    > </button>
                {/each}
            </div>
        </div>
    {/each}
</div>
