<script lang="ts">
    import { onDestroy } from 'svelte'
    import { isTerminal } from '@svelterm/core'
    import { v86Stream } from 'v86-stream'
    import type { TerminalStream } from '@svelterm/vt100'

    const { default: EmbeddedTerminal } = isTerminal()
        ? await import('@svelterm/vt100/EmbeddedTerminalRegion')
        : await import('@svelterm/vt100/EmbeddedTerminalDom')

    const stream: TerminalStream = v86Stream({ uart: 0 })
    onDestroy(() => stream.close())
</script>

<div class="shell">
    <EmbeddedTerminal {stream} />
</div>

<style>
    .shell {
        width: 100%;
        height: 100%;
        background: #000;
        padding: 8px;
        box-sizing: border-box;

        @media (display-mode: terminal) {
            padding: 1cell;
            background: default;
        }
    }
</style>
